class MySqliteDb(object):
    """Sqlite3 Db Class"""

    #构造方法（传入数据库名）
    def __init__(self, dbname="mys.db"):
        self.dbname = dbname
        self.con = None
        self.curs = None

    #定义获取数据连接的游标
    def getCursor(self):
        self.con = sqlite3.connect(self.dbname)
        if self.con:
            self.curs = self.con.cursor()

    #定义关闭数据库连接
    def closeDb(self):
        if self.curs:
            self.curs.close()
        if self.con:
            self.con.commit()
            self.con.close()

    #上下文管理器协议方法
    def __enter__(self):
        self.getCursor()
        return self.curs
        
    
    #上下文管理器协议方法
    def __exit__(self, exc_type, exc_val, exc_tb):
        if exc_val:
            print("Exception has generate: ", exc_val)
            print("Sqlite3 execute error!")
        self.closeDb()
